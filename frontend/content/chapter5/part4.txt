Display sum and average of n array elements which are accepted from a user using for Loop by function.

```cpp

 #include <iostream>

 



 using namespace std;

 



 // Function to calculate the sum and average of array elements

 void calculateSumAndAverage(int arr[], int n) {

  int sum = 0;

 



  // Calculate the sum of array elements using a for loop

  for (int i = 0; i < n; i++) {

  sum += arr[i];

  }

 



  // Calculate the average

  double average = static_cast<double>(sum) / n;

 



  // Display the sum and average

  cout << "Sum of array elements: " << sum << endl;

  cout << "Average of array elements: " << average << endl;

 }

 



 int main() {

  int n;

 



  // Accept the number of elements from the user

  cout << "Enter the number of elements: ";

  cin >> n;

 



  // Declare an array of size n

  int arr[n];

 



  // Accept array elements from the user

  cout << "Enter the array elements:" << endl;

  for (int i = 0; i < n; i++) {

  cin >> arr[i];

  }

 



  // Call the function to calculate and display the sum and average

  calculateSumAndAverage(arr, n);

 



  return 0;

 }

 ይህ የ C++ ፕሮግራም ተጠቃሚው ያስገባቸውን የ "array" አባላቶች ድምር እና አማካይ የሚያሰላ ነው።

  - `int n`: የ "array"ው መጠን (የአባላቶቹ ቁጥር) ነው።

for (int i = 0; i < n; i++)

  - ይህ "loop" እያንዳንዱን የ "array" አባል ለመጎብኘት ይጠቅማል።

  - `i = 0`: "loop"ው የሚጀምረው ከመጀመሪያው አባል ነው።

  - `i < n`: "loop"ው የሚቀጥለው `i` ከ "array"ው መጠን በታች እስካለ ድረስ ነው።

  - `i++`: በእያንዳንዱ ጊዜ `i` በ 1 ይጨምራል።

ይህ ማለት  sum=sum+arr[i] ጋ ተመሳሳይ ነው፡፡



 5. double average = static_cast<double>(sum) / n;

  - አማካዩን ለመስራት ድምሩን በ "array"ው መጠን ይካፈላል። `static_cast<double>(sum)` ድምሩን ወደ "double" ይቀይረዋል፣ ምክንያቱም አማካዩ አስርዮሽ ቁጥር ሊሆን ይችላል።

 6. cout << "Sum of array elements: " << sum << endl;

  - ይህ ፕሮግራሙ የሚጀምርበት ዋናው ተግባር ነው።

  - `int n;`: ተጠቃሚው የሚያስገባውን የ "array" መጠን ለማስቀመጥ የሚውል ተለዋዋጭ ነው።

 8. return 0;

Write a program to store the ages of six of your friends in a single array. Store each of thesix ages using the assignment operator. print the ages on the screen







```c++ code

 #include <iostream>

 using namespace std;

 int main() {

  // Declare an array to store the ages of 6 friends

  int ages[6];

   // Prompt the user to enter the ages of each friend

  cout << "Please enter the ages of your 6 friends:" << endl;

  for (int i = 0; i < 6; i++) {

  cout << "Age of friend " << i + 1 << ": ";

  cin >> ages[i]; // Store the age entered by the user

  }

  // Display the ages on the screen

  cout << "\nThe ages of your friends are:" << endl;

  for (int i = 0; i < 6; i++) {

  cout << "Friend " << i + 1 << ": " << ages[i] << " years" << endl;

  } 

  return 0;

 }

 ```

 # ማብራሪያ 

 1. #include <iostream>

  - ይሄ "iostream" የሚባለውን library ወደ ፕሮግራማችን ያስገባል:: ይሄ library "cout" እና "cin" የምትለዋን ነገር እንድንጠቀም ይረዳናል:: "cout" ማለት በ screen ላይ የሆነ ነገር ለማሳየት ስንፈልግ የምንጠቀምበት መሳሪያ ነው:: "cin" ደግሞ ተጠቃሚው ከ keyboard የሆነ ነገር እንዲያስገባልን ስንፈልግ የምንጠቀምበት መሳሪያ ነው::

  2. using namespace std;

  - ይሄን ስንጠቀም "std::" ብለን ከመጻፍ ይልቅ "cout" እና "cin" ብለን በቀላሉ መጠቀም እንችላለን::

 3. int main() {

  - ይሄ የፕሮግራሙ ዋናው ክፍል ነው:: ፕሮግራሙ እዚህ ውስጥ ነው የሚሰራው::

 4. int ages[6];

  - ይሄ "ages" የሚባል array ይፈጥራል:: Array ማለት ብዙ ነገሮችን በአንድ ላይ ማስቀመጥ የምንችልበት ሳጥን ማለት ነው:: በዚህ ሳጥን ውስጥ 6 ኢንቲጀር ቁጥሮችን ማስቀመጥ እንችላለን:: ኢንቲጀር ማለት ሙሉ ቁጥር ማለት ነው (ለምሳሌ 25, 30, 22...)::

 5. cout << "Please enter the ages of your 6 friends:" << endl;

  - ይሄ በ screen ላይ "እባክዎን የ 6 ጓደኞችዎን እድሜ ያስገቡ:" የሚል ጽሑፍ ያሳያል:: "endl" ማለት ወደ አዲሱ መስመር ውረድ ማለት ነው::

 6. for (int i = 0; i < 6; i++) { ... }
